// Dashboard functionality document.addEventListener('DOMContentLoaded', () => { // Update stats updateStats(); // Update recent activity updateRecentActivity(); // Update course progress updateCourseProgress(); }); // Update stats function updateStats() { // Get user ID const userId = getUserId(); const progress = JSON.parse(localStorage.getItem(userId + '_courseProgress')) || {}; const quizScores = JSON.parse(localStorage.getItem(userId + '_quizScores')) || {}; const achievements = JSON.parse(localStorage.getItem(userId + '_unlockedAchievements')) || []; // Calculate total lessons completed let totalLessons = 0; let totalCourses = 0; Object.keys(progress).forEach(course => { totalLessons += progress[course].length; if (progress[course].length > 0) { totalCourses++; } }); // Calculate average quiz score let totalScore = 0; let quizCount = 0; Object.keys(quizScores).forEach(quiz => { totalScore += quizScores[quiz].percentage; quizCount++; }); const averageScore = quizCount > 0 ? Math.round(totalScore / quizCount) : 0; // Update UI document.getElementById('total-lessons').textContent = totalLessons; document.getElementById('total-courses').textContent = totalCourses; document.getElementById('total-achievements').textContent = achievements.length; document.getElementById('average-score').textContent = `${averageScore}%`; } // Update recent activity function updateRecentActivity() { const container = document.getElementById('recent-activity'); // In a real app, this would come from actual user activity data const activities = [ { action: 'Completed HTML Lesson 5', time: '2 hours ago' }, { action: 'Scored 90% on CSS Quiz', time: '1 day ago' }, { action: 'Unlocked Achievement: First Steps', time: '2 days ago' }, { action: 'Started JavaScript Course', time: '3 days ago' } ]; let html = ''; activities.forEach(activity => { html += `

${activity.action}

${activity.time}
`; }); container.innerHTML = html; } // Update course progress function updateCourseProgress() { const container = document.getElementById('course-progress'); // Course data (in a real app, this would come from a data source) const courses = [ { id: 'html', name: 'HTML Fundamentals', progress: 75 }, { id: 'css', name: 'CSS Styling', progress: 40 }, { id: 'javascript', name: 'JavaScript Essentials', progress: 20 }, { id: 'python', name: 'Python Programming', progress: 0 } ]; let html = ''; courses.forEach(course => { html += `

${course.name}

${course.progress}% Complete

Continue
`; }); container.innerHTML = html; }
